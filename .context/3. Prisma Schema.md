```prisma
// schema.prisma
// Kaven Boilerplate - Backend Schema

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

/// @model Tenant
/// Represents a single customer instance in the multi-tenant system.
/// [SECURITY] Isolated via RLS policies.
model Tenant {
  id          String   @id @default(uuid())
  name        String   @unique
  subdomain   String   @unique
  created_at  DateTime @default(now())
  updated_at  DateTime @updatedAt

  // Relationships
  users       User[]
  metrics     Metric[]
  plan        Plan?   @relation(fields: [plan_id], references: [id])
  plan_id     String?

  @@map("tenants")
}

/// @model User
/// Represents a user within a specific tenant.
/// [SECURITY] Passwords must be hashed before storage. RLS ensures tenant isolation.
/// [SECURITY] The `is_adm` flag determines administrative privileges within the tenant.
model User {
  id            String   @id @default(uuid())
  tenant        Tenant   @relation(fields: [tenant_id], references: [id], onDelete: Cascade) // [ASSUMPTION] ON DELETE CASCADE for user if tenant is deleted
  tenant_id     String
  email         String   @unique
  password_hash String   /// [SECURITY] Hashed using bcrypt or argon2
  first_name    String?
  last_name     String?
  is_adm        Boolean  @default(false) /// [SECURITY] Administrative role flag
  created_at    DateTime @default(now())
  updated_at    DateTime @updatedAt

  @@map("users")
}

/// @model Plan
/// Represents a subscription plan available for tenants.
model Plan {
  id          String   @id @default(uuid())
  name        String   @unique
  description String?
  features    Json?    /// [EXAMPLE] {"max_users": 10, "storage_gb": 5}
  price       Decimal?
  created_at  DateTime @default(now())
  updated_at  DateTime @updatedAt

  // Relationships
  tenants     Tenant[]

  @@map("plans")
}

/// @model Metric
/// Stores observability metrics per tenant, aligned with PDR requirements.
model Metric {
  id           String   @id @default(uuid())
  tenant       Tenant   @relation(fields: [tenant_id], references: [id], onDelete: Cascade)
  tenant_id    String
  metric_name  String   /// [EXAMPLE] 'time_response_per_tenant', 'errors_per_tenant', 'queue_length_per_tenant', 'cpu_usage_per_tenant'
  value        Float    /// [ASSUMPTION] Using Float for numeric metric values; Decimal could be used if high precision is required for financial metrics.
  timestamp    DateTime @default(now())
  labels       Json?    /// [EXAMPLE] {"endpoint": "/api/users", "status_code": 200}

  @@map("metrics")
  @@index([tenant_id])
  @@index([metric_name])
  @@index([timestamp])
  @@index([labels]) // GIN index for JSONB queries on labels
}

```
