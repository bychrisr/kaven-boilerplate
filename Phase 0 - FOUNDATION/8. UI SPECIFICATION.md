**API Call:**
```typescript
const onSubmit = async (data: LoginFormData) => {
  try {
    setLoading(true);
    const response = await api.post('/auth/login', data);
    
    if (response.data.requiresTwoFactor) {
      // Redirect to 2FA verification
      router.push('/login/2fa', { tempToken: response.data.tempToken });
    } else {
      // Store tokens
      setAccessToken(response.data.tokens.accessToken);
      setRefreshToken(response.data.tokens.refreshToken);
      
      // Redirect to dashboard
      router.push('/dashboard');
    }
  } catch (error) {
    if (error.code === 'EMAIL_NOT_VERIFIED') {
      toast.error('Please verify your email before logging in');
    } else {
      toast.error('Invalid email or password');
    }
  } finally {
    setLoading(false);
  }
};
```

**API Calls:**
```typescript
// Load users with filters
useEffect(() => {
  const loadUsers = async () => {
    try {
      setLoading(true);
      const response = await api.get('/users', {
        params: {
          page,
          pageSize,
          search: searchQuery,
          role: roleFilter,
          emailVerified: statusFilter === 'verified' ? true : statusFilter === 'unverified' ? false : undefined,
          sortBy,
          sortOrder: sortBy.startsWith('-') ? 'desc' : 'asc'
        }
      });
      
      setUsers(response.data.data);
      setTotalPages(response.data.pagination.totalPages);
      setTotalItems(response.data.pagination.totalItems);
    } catch (error) {
      toast.error('Failed to load users');
    } finally {
      setLoading(false);
    }
  };
  
  // Debounce search
  const timeoutId = setTimeout(loadUsers, 300);
  return () => clearTimeout(timeoutId);
}, [page, pageSize, searchQuery, roleFilter, statusFilter, sortBy]);
```
